  • interface extends interface ts interface PointX {x: number} interface Point extends PointX {y: number}
  • interface extends type ts type PointX = { x: number; } interface Point extends PointX {y: number}
  • type extends type ts type PointX = { x: number; }; type PointY= { y: number; }; type Point = PointX & PointY
  • type extends interface ts type PointX = { x: number; }; interface PointY {x: number} type Point = PointX & PointY
