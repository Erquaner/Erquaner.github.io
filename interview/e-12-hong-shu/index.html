function debounce(fn, timeout) {
  let timer = null;

  return function () {
    const context = this;

    clearTimeout(timer);

    timer = setTimeout(() => {
      clearTimeout(timer);
      fn.apply(context, arguments);
    }, timeout);
  };
}

  • settimeout计时不准确怎么解决
  • 说和服务器校准时间
